// Classe estática que implementa o algoritmo DFS (Busca em Profundidade)
public static class DFS
{
    /// Executa o DFS em todo o grafo, retornando tempos de descoberta (d),
    /// tempos de finalização (f) e o predecessor de cada vértice.
    public static (int[] d, int[] f, int[] pred) Executar(GrafoListaAdjacencia grafo)
    {
        int V = grafo.Vertices;

        // Arrays para tempos de descoberta, finalização e predecessor.
        int[] d = new int[V];
        int[] f = new int[V];
        int[] pred = new int[V];

        // cor: 0 = BRANCO (não visitado), 1 = CINZA (em processamento), 2 = PRETO (finalizado).
        int[] cor = new int[V];

        // Variável global que conta os passos do tempo (incrementa a cada evento).
        int tempo = 0;

        // Inicializa todas as estruturas com -1 e cor 0 (branco).
        for (int i = 0; i < V; i++)
        {
            d[i] = -1; f[i] = -1; pred[i] = -1; cor[i] = 0;
        }

        // Função recursiva local que visita um vértice e todos os seus alcançáveis.
        void Visitar(int u)
        {
            // 1. Marca como CINZA (descoberto, mas ainda processando).
            cor[u] = 1;

            // 2. Incrementa o tempo e registra o momento da descoberta.
            d[u] = ++tempo;

            // 3. Para cada vizinho 'v' de 'u'...
            foreach (var (v, _) in grafo.ObterVizinhos(u))
            {
                // Se 'v' está BRANCO, ele se torna filho de 'u' na árvore DFS.
                if (cor[v] == 0)
                {
                    pred[v] = u;          // Define predecessor.
                    Visitar(v);           // Chama recursivamente para aprofundar a busca.
                }
                // Nota: se 'v' for CINZA, temos uma aresta de retorno (ciclo).
                // Se for PRETO, é aresta de avanço ou cruzamento.
            }

            // 4. Após processar todos os vizinhos, marca como PRETO (finalizado).
            cor[u] = 2;

            // 5. Incrementa o tempo e registra o momento da finalização.
            f[u] = ++tempo;
        }

        // Para cada vértice do grafo, se ele estiver BRANCO, inicia uma nova árvore DFS.
        // Isso garante que vértices desconexos também sejam visitados.
        for (int u = 0; u < V; u++)
        {
            if (cor[u] == 0)
                Visitar(u);
        }

        // Retorna os arrays: d (descoberta), f (finalização) e pred (predecessor).
        return (d, f, pred);
    }
}
