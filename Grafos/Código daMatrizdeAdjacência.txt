/// Grafo representado por MATRIZ DE ADJACÊNCIA.
/// Espaço: O(V²). Verificação de aresta: O(1). Listagem de vizinhos: O(V).
/// Ideal para grafos densos.

public class GrafoMatrizAdjacencia
{
    private int[,] matriz;          // matriz[origem, destino] = peso (0 se não houver aresta)
    public int Vertices { get; private set; }
    public bool Direcionado { get; private set; }

    public GrafoMatrizAdjacencia(int vertices, bool direcionado = false)
    {
        Vertices = vertices;
        Direcionado = direcionado;
        matriz = new int[vertices, vertices]; // Inicializa com 0 (sem arestas)
    }

     Adiciona aresta com peso. Se não for direcionado, adiciona nos dois sentidos. 
    public void AdicionarAresta(int origem, int destino, int peso = 1)
    {
        if (origem < 0 || origem >= Vertices || destino < 0 || destino >= Vertices)
            throw new ArgumentOutOfRangeException("Vértice inválido.");
        matriz[origem, destino] = peso;
        if (!Direcionado)
            matriz[destino, origem] = peso; // Espelho para grafos não direcionados
    }

     Verifica se existe aresta entre dois vértices (O(1)). 
    public bool ExisteAresta(int origem, int destino) => matriz[origem, destino] != 0;

     Retorna o peso da aresta (0 se inexistente). 
    public int ObterPeso(int origem, int destino) => matriz[origem, destino];

     Lista todos os vizinhos de um vértice (O(V)). 
    public int[] ObterVizinhos(int vertice)
    {
        var vizinhos = new List<int>();
        for (int i = 0; i < Vertices; i++)
            if (matriz[vertice, i] != 0)
                vizinhos.Add(i);
        return vizinhos.ToArray();
    }
}
