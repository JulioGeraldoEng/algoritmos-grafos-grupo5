// Classe estática que implementa o algoritmo BFS (Busca em Largura)
public static class BFS
{
    /// Executa o BFS a partir de um vértice fonte.
    /// Retorna as distâncias (número de arestas) e os predecessores.
    public static (int[] distancias, int[] predecessores) Executar(GrafoListaAdjacencia grafo, int fonte)
    {
        int V = grafo.Vertices;

        // Arrays de resultado: distância e predecessor de cada vértice.
        int[] dist = new int[V];
        int[] pred = new int[V];

        // Array de controle: 'visitado[v]' é true se o vértice já foi descoberto.
        bool[] visitado = new bool[V];

        // Inicializa distâncias com -1 (infinito) e predecessores com -1 (sem pai).
        for (int i = 0; i < V; i++)
        {
            dist[i] = -1;
            pred[i] = -1;
        }

        // Fila FIFO para controlar a ordem de exploração (BFS usa fila).
        Queue<int> fila = new Queue<int>();

        // Configura o vértice fonte: distância 0, marcado como visitado, e enfileirado.
        dist[fonte] = 0;
        visitado[fonte] = true;
        fila.Enqueue(fonte);

        // Enquanto houver vértices na fila, processa o próximo.
        while (fila.Count > 0)
        {
            // Remove o vértice da frente da fila (u).
            int u = fila.Dequeue();

            // Para cada vizinho 'v' de 'u' (obtido da lista de adjacência)...
            foreach (var (v, _) in grafo.ObterVizinhos(u))
            {
                // Se 'v' ainda não foi visitado, ele é descoberto agora.
                if (!visitado[v])
                {
                    visitado[v] = true;          // Marca como visitado.
                    dist[v] = dist[u] + 1;       // Distância = distância do pai + 1.
                    pred[v] = u;                 // Define 'u' como predecessor de 'v'.
                    fila.Enqueue(v);             // Coloca 'v' no final da fila para explorar depois.
                }
            }
        }

        // Retorna os arrays calculados.
        return (dist, pred);
    }
}
